dynamic lerp(num a, num b, num x) #

Performs linear interpolation between values a and b. Returns the value between a and b proportional to x (when x is between 0 and 1. When x is outside this range, the return value is a linear extrapolation).

lerp(num a, num b, num x) {
 requireArgument(isValidNumber(a), 'a');
 requireArgument(isValidNumber(b), 'b');
 requireArgument(isValidNumber(x), 'x');
 return a + x * (b - a);
}

void requireArgumentContainsPattern(Pattern pattern, String argValue, String argName) {
 if(pattern == null) {
   throw new InvalidOperationError("That's just sad. No null pattern");
 }
 requireArgumentNotNull(argValue, argName);
 if(!argValue.contains(pattern)) {
   throw new DetailedArgumentError(argName,
       'The value "$argValue" does not contain the pattern "${pattern}"');
 }
}

void requireArgumentNotNullOrEmpty(String argument, String argName) {
 _metaRequireArgumentNotNullOrEmpty(argName);
 if(argument == null) {
   throw new NullArgumentError(argName);
 }
 else if(argument.length == 0) {
   throw new DetailedArgumentError(argName, 'cannot be an empty string');
 }
}

 }
}
